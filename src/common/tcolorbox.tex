\RequirePackage{tcolorbox}

\tcbuselibrary{breakable}
\tcbuselibrary{listings}
\tcbuselibrary{skins}
\tcbuselibrary{theorems}
\tcbuselibrary{xparse}

\tcbset{
  breakable,
  colback  = tcolorbox!5,
  colframe = tcolorbox,
  enhanced,
  fonttitle = \bfseries,
}

  colback    = theorem!5,
  colframe   = theorem,
  label type = theorem,
}{thm}

\NewTcbTheorem[number within = section]{lem}{Lemma}{
  colback    = lemma!5,
  colframe   = lemma,
  label type = lemma,
}{lem}
