\subsection{\pkg{tcolorbox} - Coloured boxes, for LaTeX examples and theorems, etc}

\begin{demo}[listing side text]{}{tcolorbox}
  \begin{tcolorbox}
    This is a \textbf{tcolorbox}.
  \end{tcolorbox}
\end{demo}

\begin{demo}[listing side text]{}{theorem}
  \begin{thm}{Theorem Title}{theorem-label}
    This is \cref{thm:theorem-label}.
  \end{thm}
\end{demo}

\begin{demo}[listing side text]{}{lemma}
  \begin{lem}{Lemma Title}{lemma-label}
    This is \cref{lem:lemma-label}.
  \end{lem}
\end{demo}
